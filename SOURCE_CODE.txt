    private int boyutsay(File file) throws FileNotFoundException, IOException
    {
        BufferedReader reader = null;
        reader = new BufferedReader(new FileReader(file));
        String satir=null;
        int sayac=-1;
        do
        {
                satir = reader.readLine();
                sayac++;
        }
        while(satir!=null);
        return (int) Math.sqrt(sayac);
    }
